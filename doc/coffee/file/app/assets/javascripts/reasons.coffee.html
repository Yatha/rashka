reasons = 
{

  /*
  Отображение трех причин на главной
   */
  init: function() {
    this.get_three();
    return setInterval((function() {
      if (!app.mobile) {
        return reasons.get_three();
      }
    }), 10000);
  },

  /*
  Грузит три случайные причины в блок причин
   */
  get_three: function() {
    if (!$("body").hasClass("loading_three")) {
      return $.ajax({
        type: 'GET',
        url: "../reasons/get_three_random",
        dataType: 'script',
        beforeSend: function() {
          $("body").addClass("loading_three");
          $("#more_reasons").toggleClass("loading");
          return $("#more_reasons img").toggleClass("hidden");
        },
        success: function() {
          $("body").removeClass("loading_three");
          $("#more_reasons").toggleClass("loading");
          return $("#more_reasons img").toggleClass("hidden");
        }
      });
    }
  },

  /*
  Берет текст причины и вставляет его в поле рассказать у модульного окошка
  @param share_reason_button [DOM element] кнопка рассказать у причины
   */
  copy_reason: function(share_reason_button) {
    var text;
    text = $.trim(share_reason_button.parent().find(".text").text());
    $("#share_modal").modal();
    return $("#reason_field").val(text).trigger("change");
  }
}
